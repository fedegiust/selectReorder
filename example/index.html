selectReorder

Example 1 - With default options

$('#example_1').selectReorder();

Example 2 - Custom labels

$('#example_2').selectReorder({labelbtnUp: "Arriba", labelbtnDown: "Abajo"});

Example 3 - jQuery UI

$('#example_3').selectReorder({jQueryUI: true});

Example 4 - Buttons on the left hand side

$('#example_4').selectReorder({btnPosition: "left"});

Example 5 - Don't wrap in div

$('#example_5').selectReorder({wrapInDiv: false});
